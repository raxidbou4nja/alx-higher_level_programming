# Import the MyList class from the 1-my_list module
from 1-my_list import MyList

# Create an instance of MyList
my_list = MyList()

# Check if my_list is an instance of list
assert isinstance(my_list, list)

# Example 1: Sorting an empty list
assert my_list.print_sorted() == []

# Example 2: Adding items to the list and sorting
my_list.append(7)
my_list.append(3)
my_list.append(1)
assert my_list.print_sorted() == [1, 3, 7]

# Example 3: Adding zero values and sorting
my_list.append(0)
assert my_list.print_sorted() == [0, 1, 3, 7]

# Example 4: Attempting to add infinity (not supported)
try:
    my_list.append(float('inf'))
except ValueError:
    pass  # This should raise a ValueError

# Example 5: Adding negative values and sorting
my_list.append(-15)
assert my_list.print_sorted() == [-15, 0, 1, 3, 7]

# Example 6: Repeating elements and sorting
my_list.append(-15)
my_list.append(0)
my_list.append(7)
assert my_list.print_sorted() == [-15, -15, 0, 0, 1, 3, 7, 7]

# Example 7: All elements in the list are the same
my_list.clear()
my_list.append(5)
my_list.append(5)
my_list.append(5)
my_list.append(5)
my_list.append(5)
assert my_list.print_sorted() == [5, 5, 5, 5, 5]

# Example 8: Adding a large integer and sorting
my_list.append(10000000000000000000000000000000000000000000)
assert my_list.print_sorted() == [5, 5, 5, 5, 5, 10000000000000000000000000000000000000000000]

# Example 9: Attempting to add None (not supported)
try:
    my_list.append(None)
except TypeError:
    pass  # This should raise a TypeError

# Example 10: Attempting to add NULL (NameError because NULL is not defined)
try:
    my_list.append(NULL)
except NameError:
    pass  # This should raise a NameError
